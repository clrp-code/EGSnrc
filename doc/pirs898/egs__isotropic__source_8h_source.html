253  void getPositionDirection(EGS_RandomGenerator *rndm,
254  EGS_Vector &x, EGS_Vector &u, EGS_Float &wt) {
255  bool ok = true;
256  do {
257  x = shape->getRandomPoint(rndm);
258  if (geom) {
259  if (gc == IncludeAll) {
260  ok = geom->isInside(x);
261  }
262  else if (gc == ExcludeAll) {
263  ok = !geom->isInside(x);
264  }
265  else if (gc == IncludeSelected) {
266  ok = false;
267  int ireg = geom->isWhere(x);
268  for (int j=0; j<nrs; ++j) {
269  if (ireg == regions[j]) {
270  ok = true;
271  break;
272  }
273  }
274  }
275  else {
276  ok = true;
277  int ireg = geom->isWhere(x);
278  for (int j=0; j<nrs; ++j) {
279  if (ireg == regions[j]) {
280  ok = false;
281  break;
282  }
283  }
284  }
285  }
286  }
287  while (!ok);
288 
289  u.z = rndm->getUniform()*(buf_1 - buf_2) - buf_1;
290 
291  //u.z = 2*rndm->getUniform()-1;
292  EGS_Float sinz = 1-u.z*u.z;
293  if (sinz > epsilon) {
294  sinz = sqrt(sinz);
295  EGS_Float cphi, sphi;
296  //rndm->getAzimuth(cphi,sphi);
297  // sample phi, slower than rndm->getAzimuth
298  EGS_Float phi = min_phi +(max_phi - min_phi)*rndm->getUniform();
299  cphi = cos(phi);
300  sphi = sin(phi);
301  u.x = sinz*cphi;
302  u.y = sinz*sphi;
303  }
304  else {
305  u.x = 0;
306  u.y = 0;
307  }
308  wt = 1;
309  };
