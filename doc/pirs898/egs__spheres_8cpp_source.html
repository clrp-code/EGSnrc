90 int EGS_cSpheres::inside(const EGS_Vector &x) {
91 
92  EGS_Vector tmp(x-xo);
93  EGS_Float r_sq=tmp.length2();
94 
95  // are we outside off all spheres? If so return that region number
96  if (r_sq>R2[nreg-1]) {
97  return -1;
98  }
99 
100  /*
101  * algorithm below fails for particle in central sphere ...ugh!
102  */if (r_sq<R2[0]) {
103  return 0;
104  }
105 
106  // search for region containing particle
107  int is=0,os=nreg,ms;
108  while (os-is>1) {
109  ms=(is+os)/2;
110  if (r_sq<=R2[ms]) {
111  os=ms;
112  }
113  else {
114  is=ms;
115  }
116  }
117  return os;
118 }

132 EGS_Float EGS_cSpheres::howfarToOutside(int ireg, const EGS_Vector &x,
133  const EGS_Vector &u) {
134  if (ireg < 0) {
135  return 0;
136  }
137  EGS_Vector xp(x - xo);
138  EGS_Float aa = xp*u, aa2 = aa*aa;
139  EGS_Float bb2 = xp.length2();
140  EGS_Float R2b2 = R2[nreg-1] - bb2;
141  if (R2b2 <= 0) {
142  return 0; // outside within precision
143  }
144  EGS_Float tmp = sqrt(aa2 + R2b2);
145  return aa > 0 ? R2b2/(tmp + aa) : tmp - aa;
146 }

148 int EGS_cSpheres::howfar(int ireg, const EGS_Vector &x,
149  const EGS_Vector &u, EGS_Float &t, int *newmed, EGS_Vector *normal) {
150  int direction_flag=-1; /* keep track of direction entering or exiting a
151  sphere boundary */
152  double d=1e35; // set a maximum distance from a boundary
153 
154  EGS_Vector xp(x - xo);
155  double aa = xp*u, aa2 = aa*aa;
156  double bb2 = xp.length2();
157 
158  double rad, R2b2, tmp;
159 
160  // check if we are inside of any regions at all? ...
161  if (ireg>=0) {
162 
163  /* check if particle is moving towards or away from the sphere(s) centre.
164  * we loose here if the particle is in the centre sphere as we don't need
165  * to check this condition. see next 'if' statement
166  */
167  if (aa >= 0 || !ireg) {
168 
169  /* ie. particle moving away from center of
170  spheres, OR it is IN the innermost sphere
171  => we must check the outer sphere only
172  */
173  R2b2 = R2[ireg] - bb2;
174  if (R2b2 <= 0 && aa > 0) {
175  d = boundaryTolerance; // hopefully a truncation problem
176  }
177  else {
178  tmp = aa2 + R2b2;
179  if (tmp > 0) {
180  tmp = sqrt(tmp);
181  }
182  else {
183  if (tmp < -boundaryTolerance) {
184  egsWarning("EGS_cSpheres::howfar: something is wrong\n");
185  egsWarning(" we think we are in region %d, but R2b2=%g",
186  ireg,R2b2);
187  }
188  tmp = 0;
189  }
190  d = aa > 0 ? R2b2/(tmp + aa) : tmp - aa;
191  // the above reduces roundoff, which is significant
192  // when aa2 is large compared to R2b2 and aa>0
193  }
194  rad = -R[ireg];
195  direction_flag=ireg+1;
196  if (direction_flag >= nreg) {
197  direction_flag = -1;
198  }
199  }
200  else {
201 
202  /* so now we know the particle is moving towards the centre of the
203  * spheres. check to see if its trajectory will intersect the nested
204  * sphere - we are guaranteed there is one - we checked that already!
205  */
206  R2b2 = R2[ireg-1] - bb2;
207  tmp = aa2 + R2b2;
208  if (tmp <= 0) { // we will not intersect the nested sphere
209  R2b2 = R2[ireg] - bb2;
210  tmp = aa2 + R2b2;
211  if (tmp > 0) {
212  d = sqrt(tmp) - aa;
213  }
214  else {
215  d = -aa;
216  }
217  rad = -R[ireg];
218  direction_flag=ireg+1;
219  if (direction_flag >= nreg) {
220  direction_flag = -1;
221  }
222  }
223  else {
224  // we're hitting the inner sphere (from the outside)
225  tmp = sqrt(tmp);
226  d = -R2b2/(tmp - aa);
227  direction_flag=ireg-1;
228  rad = R[direction_flag];
229  }
230  }
231  }
232  else {
233  // we are not inside any of the spherical regions of interest
234  if (aa<0) { // we _might_ intersect the largest sphere
235  R2b2 = R2[nreg-1] - bb2;
236  tmp = aa2 + R2b2;
237  if (tmp > 0) { // we *will* intersect the largest sphere
238  d = -R2b2/(sqrt(tmp) - aa);
239  direction_flag=nreg-1;
240  rad = R[direction_flag];
241  }
242  }
243  }
244 
245 #ifdef SPHERES_DEBUG
246  if (isnan(d)) {
247  egsWarning("\nGot nan\n");
248  }
249 
250  if (d < -boundaryTolerance) {
251  egsWarning("\nNegative step?: %g\n",d);
252  egsWarning("ireg=%d inew=%d aa=%g bb2=%g\n",ireg,direction_flag,aa,bb2);
253  //exit(1);
254  }
255 
256  last_x = x;
257  last_u = u;
258  last_ireg = ireg;
259  last_d = d;
260  last_t = t;
261  last_aa = aa;
262  last_bb2 = bb2;
263  last_R2b2 = R2b2;
264  last_tmp = tmp;
265 #endif
266 
267  // check desired step size against this d
268  if (d<=t) {
269  t=d;
270  if (newmed) if (direction_flag >= 0) {
271  *newmed = medium(direction_flag);
272  }
273  else {
274  *newmed = -1;
275  }
276  if (normal) {
277  EGS_Vector n(xp + u*d);
278  *normal = n*(1/rad);
279  }
280  return direction_flag;
281  }
282  return ireg;
283 }

286 EGS_Float EGS_cSpheres::hownear(int ireg, const EGS_Vector &x) {
287  EGS_Vector xp(x-xo);
288  EGS_Float r=xp.length();
289  //EGS_Float r_sq=x.x*x.x+x.y*x.y+x.z*x.z;
290  EGS_Float d;
291 
292  if (ireg>=0) {
293  d=R[ireg]-r;
294  if (ireg) {
295  EGS_Float dd=r-R[ireg-1];
296  if (dd<d) {
297  d=dd;
298  }
299  }
300  }
301  else {
302  d=r-R[nreg-1];
303  }
304 
305  return d;
