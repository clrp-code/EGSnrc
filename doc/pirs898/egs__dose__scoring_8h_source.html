30 # A general dose calculation tool. Since it is outside the scope of the
31 # EGS_Application class, a few utility methods are needed in EGS_Application
32 # to retrieve information regarding media, geometry and source.
33 #
34 # Dose is output for each dose region without any spacial information. For a
35 # more detailed dose scoring grid it is better to create a user code which
36 # could produce 3D dose distributions and graph data.
37 #
38 # Can be used in any C++ user code by entering the proper input block in
39 # the ausgab object definition block.
40 #
41 # Prints out dose in each dose scoring region.
42 #
43 # Dose scoring regions can be defined by input using one the following
44 # syntax for individual regions:
45 #
46 # :start ausgab object definition:
47 # :start ausgab object:
48 # library = egs_dose_scoring
49 # name = some_name
50 # volume = V0 V1 ... Vn
51 # dose regions = IR0 IR1 ... IRn
52 # :stop ausgab object:
53 # :stop ausgab object definition:
54 #
55 # or by groups of consecutive regions:
56 #
57 # :start ausgab object definition:
58 # :start ausgab object:
59 # library = egs_dose_scoring
60 # name = some_name
61 # volume = V0 V1 ... Vn
62 # dose start region = IRI_0 ... IRI_N
63 # dose stop region = IRE_0 ... IRE_N
64 # :stop ausgab object:
65 # :stop ausgab object definition:
66 #
67 # where one can have same number of volume entries as group of regions (n=N)
68 # or same number of volume entries as number of individual regions. If there
69 # are more dose scoring regions than volume entries, a default volume is use for
70 # those regions without volume information. This default can be either the first
71 # volume entry or 1 g/cm3.
72 #
73 # TODO:
74 #
75 # - Classify dose as contributed by primary or scattered particles
76 # - Custom classification using latch
77 # - Dose to medium calculation
78 #
79 ###############################################################################
80 */

174  int processEvent(EGS_Application::AusgabCall iarg) {
175 
176  int ir = app->top_p.ir, imed = ir>=0 ? app->getMedium(ir):-1;
177  EGS_Float edep = app->getEdep();
178 
179  /**** energy deposition in a medium ***/
180  if (iarg <= 4 && imed >= 0 && edep > 0 && doseM) {
181  doseM->score(imed, edep*app->top_p.wt);
182  }
183 
184  /*** Check if scoring in current region ***/
185  if (dose) {
186  if (d_reg_index[ir]<0) {
187  return 0;
188  }
189  }
190 
191  /**** energy deposition in current region ***/
192  if (iarg <= 4 && ir >= 0 && edep > 0 && dose) {
193  dose->score(d_reg_index[ir], edep*app->top_p.wt);
194  }
195  return 0;
196  };
