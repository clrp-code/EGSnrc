175 EGS_I64 EGS_BeamSource::getNextParticle(EGS_RandomGenerator *, int &q,
176  int &latch, EGS_Float &E, EGS_Float &wt, EGS_Vector &x, EGS_Vector &u) {
177  if (n_reuse_photon > 0 && i_reuse_photon < n_reuse_photon) {
178  q = q_save;
179  latch = latch_save;
180  E = E_save;
181  wt = wt_save;
182  x = x_save;
183  u = u_save;
184  ++i_reuse_photon;
185  return count;
186  }
187  if (n_reuse_electron > 0 && i_reuse_electron < n_reuse_electron) {
188  q = q_save;
189  latch = latch_save;
190  E = E_save;
191  wt = wt_save;
192  x = x_save;
193  u = u_save;
194  ++i_reuse_electron;
195  return count;
196  }
197  EGS_Float te,tx,ty,tz,tu,tv,tw,twt;
198  int tq,tlatch,tiphat;
199  bool ok;
200  do {
201  sample(&te,&tx,&ty,&tz,&tu,&tv,&tw,&twt,&tq,&tlatch,&count,&tiphat);
202  //egsInformation("EGS_BeamSource::getNextParticle: Got E=%g q=%d wt=%g"
203  // " x=(%g,%g,%g) latch=%d count=%lld\n",te,tq,twt,tx,ty,tz,
204  // tlatch,count);
205  if (tq) {
206  te -= EGS_Application::activeApplication()->getRM();
207  }
208  ok = true;
209  if (te > Emax) {
210  ok = false;
211  } //egsInformation("Emax rejection\n"); }
212  if (particle_type < 2 && tq != particle_type) {
213  ok = false;
214  } // egsInformation("charge rejection"); }
215  if (particle_type == 3 && !tq) {
216  ok = false;
217  }
218  if (tx < Xmin || tx > Xmax || ty < Ymin || ty > Ymax) {
219  ok = false;
220  } // egsInformation("cutout rejection\n"); }
221  if (twt < wmin || twt > wmax) {
222  ok = false; // egsInformation("weight rejection\n");
223  }
224  }
225  while (!ok);
226  i_reuse_electron = n_reuse_electron;
227  i_reuse_photon = n_reuse_photon;
228  //egsInformation("returning particle\n");
229  E = te;
230  q = tq;
231  latch = 0; //latch = tlatch;
232  bool save_it = false;
233  if (n_reuse_photon > 1 && !tq) {
234  twt /= n_reuse_photon;
235  i_reuse_photon = 1;
236  save_it = true;
237  }
238  if (n_reuse_electron > 1 && tq) {
239  twt /= n_reuse_electron;
240  i_reuse_electron = 1;
241  save_it = true;
242  }
243  wt = twt;
244  x = EGS_Vector(tx,ty,tz);
245  u = EGS_Vector(tu,tv,tw);
246  if (save_it) {
247  q_save = tq;
248  latch_save = 0;
249  E_save = E;
250  wt_save = twt;
251  x_save = x;
252  u_save = u;
253  }
254  return count;
255 }
